public class InvalidCountryException extends Exception {

   
    public InvalidCountryException() {
        super();
    }

 
    public InvalidCountryException(String message) {
        super(message);
    }
}
public class assignment65 {

  
    public static void registerUser(String username, String userCountry) throws InvalidCountryException {
        if (!userCountry.equalsIgnoreCase("India")) {
            throw new InvalidCountryException("User Outside India cannot be registered");
        } else {
            System.out.println("User registration done successfully");
        }
    }

    public static void main(String[] args) {
       
        System.out.println("Input: Mickey, US");
        try {
            registerUser("Mickey", "US");
        } catch (InvalidCountryException e) {
            System.out.println("Output: " + e.getMessage());
        }

        System.out.println("--------------------");

       
        System.out.println("Input: Mini, India");
        try {
            registerUser("Mini", "India");
        } catch (InvalidCountryException e) {
            System.out.println("Output: " + e.getMessage());
        }
    }
}
