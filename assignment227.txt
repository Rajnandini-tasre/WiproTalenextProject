public void deleteFlowerById(int flowerId) {
    Transaction transaction = null;
    try (Session session = sessionFactory.openSession()) {
        transaction = session.beginTransaction();

        // Retrieve the flower object
        Flower flower = session.get(Flower.class, flowerId);

        if (flower != null) {
            // If the flower exists, delete it
            session.delete(flower);
            System.out.println("Flower with ID " + flowerId + " has been deleted.");
        } else {
            System.out.println("Error: No Flower with ID " + flowerId + " exists.");
        }

        transaction.commit();
    } catch (Exception e) {
        if (transaction != null) {
            transaction.rollback();
        }
        e.printStackTrace();
    }
}
