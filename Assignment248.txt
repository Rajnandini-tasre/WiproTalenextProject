Create a form which takes 3 inputs from User, Science Marks, Math Marks, English Marks. After entering the data, when the user clicks on the submit button, it has to calculate the sum of these 3 numbers and show the total marks back to the user using Spring MVC for this solution. Use XML configuration.

4.MarksController.java

package com.example.controller;

import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;

@Controller
public class MarksController {

    @RequestMapping("/marksForm")
    public String showForm() {
        return "inputForm"; // JSP page with form
    }

    @RequestMapping("/calculateMarks")
    public String calculateMarks(
            @RequestParam("science") int science,
            @RequestParam("math") int math,
            @RequestParam("english") int english,
            Model model) {

        int total = science + math + english;
        model.addAttribute("totalMarks", total);
        return "result"; // JSP page to show total
    }
}
