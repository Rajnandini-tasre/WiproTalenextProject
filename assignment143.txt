CREATE TABLE EMP (
    ID NUMBER(7) PRIMARY KEY,
    LAST_NAME VARCHAR2(25),
    FIRST_NAME VARCHAR2(25),
    DEPT_ID NUMBER(7),
    CONSTRAINT fk_emp_dept FOREIGN KEY (DEPT_ID)
        REFERENCES DEPT (Dept_ID)
);
