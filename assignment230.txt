import org.hibernate.Session;
import org.hibernate.SessionFactory;
import org.hibernate.Transaction;
import org.hibernate.cfg.Configuration;

public class Main {
    public static void main(String[] args) {
        SessionFactory sessionFactory = new Configuration()
                .configure("hibernate.cfg.xml")
                .addAnnotatedClass(CarDetails.class)
                .buildSessionFactory();

        Session session = sessionFactory.openSession();
        Transaction transaction = null;

        try {
            transaction = session.beginTransaction();

            // Create and save the CarDetails objects
            CarDetails car1 = new CarDetails("KL-07", "AB 123 Polo", "White", "Volkswagen");
            CarDetails car2 = new CarDetails("KL-07", "AB 234 Vento", "Black", "Volkswagen");
            CarDetails car3 = new CarDetails("KL-07", "AC 345 Corolla", "Silver", "Toyota");

            session.save(car1);
            session.save(car2);
            session.save(car3);
            
            transaction.commit();
            System.out.println("Records inserted successfully.");

        } catch (Exception e) {
            if (transaction != null) {
                transaction.rollback();
            }
            e.printStackTrace();
        } finally {
            session.close();
            sessionFactory.close();
        }
    }
}
