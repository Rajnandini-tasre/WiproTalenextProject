import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class assignment146 {
    // JDBC URL for Oracle database
    static final String DB_URL = "jdbc:oracle:thin:@localhost:1521:ORCL";

    // Database credentials
    static final String USER = "hr";
    static final String PASS = "hr";

    public static void main(String[] args) {
        Connection conn = null;
        Statement stmt = null;
        ResultSet rs = null;

        try {
            System.out.println("Connecting to database...");
            // Open a connection
            conn = DriverManager.getConnection(DB_URL, USER, PASS);

            System.out.println("Creating statement...");
            stmt = conn.createStatement();

            // SQL query to select all rows from the emp table
            String sql = "SELECT empno, ename FROM emp";
            rs = stmt.executeQuery(sql);

            // Print the header for the output
            System.out.println("\nEmployee Details:");
            System.out.println("---------------------------------");
            System.out.printf("%-10s %-20s\n", "EMPNO", "ENAME");
            System.out.println("---------------------------------");

            // Loop through the result set and display the data
            while (rs.next()) {
                // Retrieve by column index (1 for empno)
                int empNo = rs.getInt(1);

                // Retrieve by column name (ename)
                String eName = rs.getString("ename");

                // Display values
                System.out.printf("%-10d %-20s\n", empNo, eName);
            }
            System.out.println("---------------------------------");

        } catch (SQLException se) {
            // Handle errors for JDBC
            System.err.println("A database error occurred: " + se.getMessage());
            se.printStackTrace();
        } finally {
            // Close resources in a finally block to ensure they are always closed
            try {
                if (rs != null) rs.close();
            } catch (SQLException se) {
                se.printStackTrace();
            }
            try {
                if (stmt != null) stmt.close();
            } catch (SQLException se) {
                se.printStackTrace();
            }
            try {
                if (conn != null) conn.close();
            } catch (SQLException se) {
                se.printStackTrace();
            }
        }
    }
}
