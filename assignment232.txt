import org.hibernate.Session;
import org.hibernate.SessionFactory;
import org.hibernate.query.Query;
import java.util.List;

public class HQLQueries {
    public void displayRegNoAndManufacturer(SessionFactory sessionFactory) {
        Session session = sessionFactory.openSession();
        try {
            // HQL query to select specific properties
            String hql = "select regNo, manufacturer from CarDetails";
            Query<Object[]> query = session.createQuery(hql, Object[].class);
            List<Object[]> results = query.list();

            System.out.println("Registration Number and Manufacturer:");
            for (Object[] result : results) {
                String regNo = (String) result[0];
                String manufacturer = (String) result[1];
                System.out.println("RegNo: " + regNo + ", Manufacturer: " + manufacturer);
            }
        } finally {
            session.close();
        }
    }
}
